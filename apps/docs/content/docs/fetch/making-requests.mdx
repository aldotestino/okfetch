## Path params and query strings

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("/users/:id", {
  baseURL: "https://api.example.com",
  params: { id: 42 },
  query: { include: ["profile", "teams"], verbose: true },
});
```

Use `params` for `:id` placeholders and `query` for the query string.

## JSON body

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/posts", {
  method: "POST",
  body: {
    published: false,
    title: "Hello",
  },
});
```

## Bearer auth

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/me", {
  auth: {
    type: "bearer",
    token: process.env.API_TOKEN ?? "",
  },
});
```

## Basic auth

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/admin", {
  auth: {
    type: "basic",
    username: process.env.API_USER ?? "",
    password: process.env.API_PASSWORD ?? "",
  },
});
```

## Custom auth prefix

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/me", {
  auth: {
    type: "custom",
    prefix: "Token",
    value: process.env.API_TOKEN ?? "",
  },
});
```

## Timeout

```ts twoslash
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/health", {
  timeout: 3_000,
});
```

Use `timeout` when the caller should fail fast instead of waiting for the upstream.

## Custom `fetch`

```ts twoslash
declare const customFetch: (...args: any[]) => Promise<Response>;
// ---cut---
import { okfetch } from "@okfetch/fetch";

const result = await okfetch("https://api.example.com/health", {
  fetch: customFetch,
});
```
