## One endpoint tree

The same tree can describe route params, query strings, request bodies, success payloads, error payloads, request defaults, and streaming endpoints.

```ts twoslash
import { createEndpoints } from "@okfetch/api";
import { z } from "zod/v4";

const todoSchema = z.object({
  completed: z.boolean(),
  id: z.number(),
  title: z.string(),
  userId: z.number(),
});

const endpoints = createEndpoints({
  todos: {
    get: {
      method: "GET",
      path: "/todos/:id",
      params: z.object({ id: z.number() }),
      output: todoSchema,
    },
    list: {
      method: "GET",
      path: "/todos",
      query: z.object({
        completed: z.boolean().optional(),
      }),
      output: z.array(todoSchema),
    },
    create: {
      method: "POST",
      path: "/todos",
      body: z.object({
        title: z.string().min(1),
        userId: z.number(),
      }),
      output: todoSchema,
      error: z.object({
        code: z.string(),
        message: z.string(),
      }),
      requestOptions: {
        timeout: 5_000,
      },
    },
    events: {
      method: "GET",
      path: "/todos/events",
      stream: true,
      output: z.object({
        id: z.number(),
        message: z.string(),
      }),
    },
  },
});
```

## What the client looks like

```ts twoslash
import { createApi, createEndpoints } from "@okfetch/api";
import { z } from "zod/v4";

const endpoints = createEndpoints({
  todos: {
    get: {
      method: "GET",
      path: "/todos/:id",
      params: z.object({ id: z.number() }),
    },
    list: {
      method: "GET",
      path: "/todos",
      query: z.object({
        completed: z.boolean().optional(),
      }),
    },
    create: {
      method: "POST",
      path: "/todos",
      body: z.object({
        title: z.string().min(1),
        userId: z.number(),
      }),
    },
  },
});

const api = createApi({
  baseURL: "https://api.example.com",
  endpoints,
});

const item = await api.todos.get({ params: { id: 1 } });
const list = await api.todos.list({ query: { completed: true } });
const created = await api.todos.create({ body: { title: "Ship docs", userId: 1 } });
```

If an endpoint has no `body`, `params`, or `query` schema, the generated method can be called with request overrides only.
