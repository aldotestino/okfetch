## Typical request

Use this as the default shape for a normal typed request.

```ts twoslash
import { okfetch } from "@okfetch/fetch";
import { z } from "zod/v4";

const userSchema = z.object({
  id: z.number(),
  name: z.string(),
});

const result = await okfetch("/users/:id", {
  baseURL: "https://api.example.com",
  params: { id: 42 },
  query: { include: ["teams"] },
  outputSchema: userSchema,
});

result.match({
  err: (error) => console.error(error._tag, error.message),
  ok: (user) => console.log(user.name),
});
```
